create table "APP".FILIERE
(

create table "APP".ENSEIGNANT
(
	NUMENS INTEGER not null primary key GENERATED BY DEFAULT AS IDENTITY,

	create table "APP".GROUPE

	NUMFILIERE INTEGER not null,

create table "APP".MATIERE
(
	NUMMATIERE INTEGER not null primary key GENERATED BY DEFAULT AS IDENTITY,
	NOMMATIERE VARCHAR(255)
)espace

create table "APP".GROUPEMATIERE
(
	NUMMATIERE INTEGER not null,
	NUMGROUPE INTEGER not null,
	DATEDEBUT DATE not null,
	DATEFIN DATE not null,
	primary key (NUMMATIERE, NUMGROUPE),

	ON DELETE CASCADE ON UPDATE RESTRICT,
	CONSTRAINT GroupeMAT_Matiere_FK FOREIGN KEY (NUMMATIERE) REFERENCES  APP.MATIERE(NUMMATIERE)
	ON DELETE CASCADE ON UPDATE RESTRICT
)espace

create table "APP".SEANCE
(
	NUMSEANCE INTEGER not null primary key GENERATED BY DEFAULT AS IDENTITY,
	NUMCRENEAU INTEGER not null,
	NUMENS INTEGER not null,
	NUMMATIERE INTEGER,
	NUMGROUPE INTEGER,
	ETATSEANCE INTEGER,

	CONSTRAINT Seance_Mat_FK FOREIGN KEY (NUMMATIERE) REFERENCES  APP.MATIERE(NUMMATIERE)
	ON DELETE CASCADE ON UPDATE RESTRICT,
	CONSTRAINT Seance_Groupe_FK FOREIGN KEY (NUMGROUPE) REFERENCES  APP.GROUPE(NUMGROUPE)
	ON DELETE CASCADE ON UPDATE RESTRICT,
	CONSTRAINT Seance_Creneau_FK FOREIGN KEY (NUMCRENEAU) REFERENCES  APP.CRENEAU(NUMCRENEAU)
	ON DELETE CASCADE ON UPDATE RESTRICT,
	CONSTRAINT Seance_Enseignant_FK FOREIGN KEY (NUMENS) REFERENCES  APP.ENSEIGNANT(NUMENS)
	ON DELETE CASCADE ON UPDATE RESTRICT
